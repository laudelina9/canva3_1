    <script>
        const validCodes = ["DOCENTE-PRUEBA"];
        for(let i = 1; i <= 40; i++) {
            validCodes.push("ALUM-" + String(i).padStart(2, '0'));
        }

        // Se reemplazaron las rutas locales por URLs públicas estables e independientes de la computadora local
        const quizData = [
            {
                question: "¿Qué menú de Canva funciona como el 'centro de control' principal desde donde organizás tu trabajo, ves tus diseños recientes y accedés a plantillas recomendadas, pero no es el espacio de edición de un archivo?",
                options: ["Proyectos", "Inicio", "Elementos", "Descargar"],
                correct: 1,
                imageSrc: "https://picsum.photos/id/180/600/400", 
                imageAlt: "Pista: Panel de inicio y organización central de archivos"
            },
            {
                question: "Al finalizar un trabajo y querer guardarlo en tu dispositivo con la máxima calidad gráfica para imágenes o ilustraciones, ¿cuál es el formato que la plataforma te resalta como 'Sugerencia'?",
                options: ["PDF estándar", "JPG", "PNG", "SVG"],
                correct: 2,
                imageSrc: "https://picsum.photos/id/250/600/400",
                imageAlt: "Pista: Opciones de exportación de formatos gráficos"
            },
            {
                question: "Si estás editando un texto u objeto en el lienzo y necesitás modificar su orden de superposición (por ejemplo, enviarlo detrás de una foto o traerlo adelante), ¿qué botón de la barra flotante debés presionar?",
                options: ["Agrupar", "Duplicar", "Posición", "Bloquear contenido"],
                correct: 2,
                imageSrc: "https://picsum.photos/id/367/600/400",
                imageAlt: "Pista: Herramientas de ordenamiento y capas organizativas"
            },
            {
                question: "Hiciste clic sobre una fotografía dentro del diseño y querés variar parámetros como brillo, contraste o aplicarle filtros predefinidos. ¿Qué opción de la barra superior abre este menú?",
                options: ["Editar imagen", "Texto Mágico", "Elementos", "Redimensionar"],
                correct: 0,
                imageSrc: "https://picsum.photos/id/486/600/400",
                imageAlt: "Pista: Configuración avanzada de elementos visuales y efectos"
            }
        ];

        let currentQuestionIdx = 0;
        let score = 0;
        let currentCode = "";
        let timerInterval = null;
        let timeLeft = 240; 
        let blurCount = 0;
        let answersLog = [];
        let isQuizEnded = false;

        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        function playAlarmBuzzer() {
            try {
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(850, audioCtx.currentTime);
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                gain.gain.setValueAtTime(0.15, audioCtx.currentTime);
                osc.start();
                gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.6);
                osc.stop(audioCtx.currentTime + 0.6);
            } catch(e) {}
        }

        function speakWarning() {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel(); 
                const msg = new SpeechSynthesisUtterance();
                msg.text = "Atención. Has salido del examen. Regresa inmediatamente al juego o tu prueba será anulada.";
                msg.lang = 'es-AR'; 
                msg.rate = 0.95; 
                window.speechSynthesis.speak(msg);
            }
        }

        window.addEventListener('blur', () => {
            if (isQuizEnded || currentCode === "" || currentCode === "DOCENTE-PRUEBA") return;
            
            blurCount++;
            document.getElementById("cheat-indicator").innerText = `Desvíos: ${blurCount} / 2`;
            
            playAlarmBuzzer();
            speakWarning();
            
            if (blurCount === 1) {
                alert("⚠️ DETECCIÓN DE ACCIÓN EXTERNA: Saliste de la pantalla de evaluación. Recordá que no podés buscar en Google ni usar IA. Primer aviso.");
            } else if (blurCount === 2) {
                alert("⚠️ ÚLTIMA ADVERTENCIA: Si salís nuevamente del entorno, el sistema ejecutará el cierre automático asignando nota cero.");
            } else if (blurCount >= 3) {
                triggerAutoFailByCheat();
            }
        });

        function triggerAutoFailByCheat() {
            isQuizEnded = true;
            clearInterval(timerInterval);
            
            alert("❌ EXAMEN ANULADO: Infracción reiterada a las normas de seguridad. Tu examen se cierra con nota 0.");
            
            document.getElementById("screen-quiz").classList.remove("active");
            document.getElementById("screen-finish").classList.add("active");

            document.getElementById("form-code").value = currentCode;
            document.getElementById("form-correct").value = 0;
            document.getElementById("form-incorrect").value = quizData.length;
            document.getElementById("form-grade").value = "1 / 4 (ANULADO POR TRAMPA)";
            document.getElementById("form-status").value = "ANULADO_POR_INTENTO_COPIA";
            document.getElementById("form-details").value = "El software bloqueó la prueba porque el alumno minimizó la ventana, intentó realizar capturas o usar inteligencia artificial 3 veces.";
            
            document.getElementById("formspree-form").submit();
        }

        function validateLogin() {
            const inputField = document.getElementById("student-code");
            const code = inputField.value.trim().toUpperCase();
            const alertBox = document.getElementById("login-alert");

            if (validCodes.includes(code)) {
                currentCode = code;
                alertBox.style.display = "none";
                
                document.getElementById("screen-login").classList.remove("active");
                document.getElementById("screen-quiz").classList.add("active");
                
                loadQuestion();
            } else {
                alertBox.style.display = "block";
            }
        }

        function startTimer() {
            clearInterval(timerInterval);
            timeLeft = 240;
            updateTimerDisplay();

            timerInterval = setInterval(() => {
                timeLeft--;
                updateTimerDisplay();

                if(timeLeft <= 20) {
                    document.getElementById("timer-display").classList.add("timer-warn");
                } else {
                    document.getElementById("timer-display").classList.remove("timer-warn");
                }

                if(timeLeft <= 0) {
                    clearInterval(timerInterval);
                    handleTimeOut();
                }
            }, 1000);
        }

        function updateTimerDisplay() {
            const min = Math.floor(timeLeft / 60);
            const sec = timeLeft % 60;
            document.getElementById("timer-display").innerText = `⏱️ ${min < 10 ? '0':''}${min}:${sec < 10 ? '0':''}${sec}`;
        }

        function handleTimeOut() {
            alert("⏰ Tiempo límite agotado (4 minutos). Esta pregunta se procesa como incorrecta.");
            const currentQuestion = quizData[currentQuestionIdx];
            const buttons = document.querySelectorAll('.option-btn');
            
            buttons.forEach(b => b.disabled = true);
            buttons[currentQuestion.correct].classList.add('correct');
            
            answersLog.push({ questionId: currentQuestionIdx + 1, selection: "TIEMPO_EXPIRADO", result: "INCORRECTA" });
            document.getElementById("next-btn").disabled = false;
        }

        function loadQuestion() {
            if(isQuizEnded) return;
            const q = quizData[currentQuestionIdx];
            
            document.getElementById("question-counter").innerText = `Pregunta ${currentQuestionIdx + 1} de ${quizData.length}`;
            document.getElementById("progress-bar").style.width = `${(currentQuestionIdx / quizData.length) * 100}%`;
            document.getElementById("next-btn").disabled = true;

            document.getElementById("quiz-dynamic-content").innerHTML = `
                <div class="question-text">${q.question}</div>
                <div class="content-layout">
                    <div class="image-wrapper">
                        <span>🔍 Captura de pantalla de la guía (Pista)</span>
                        <img src="${q.imageSrc}" alt="${q.imageAlt}" class="hint-image">
                    </div>
                    <div>
                        <ul class="options-list">
                            ${q.options.map((opt, i) => `
                                <li><button class="option-btn" onclick="processAnswer(${i}, this)">${String.fromCharCode(65 + i)}) ${opt}</button></li>
                            `).join('')}
                        </ul>
                    </div>
                </div>
            `;
            startTimer();
        }

        function processAnswer(selectedIndex, clickedBtn) {
            clearInterval(timerInterval);
            const q = quizData[currentQuestionIdx];
            const buttons = document.querySelectorAll('.option-btn');

            buttons.forEach(b => b.disabled = true);

            let resultStatus = "INCORRECTA";
            if (selectedIndex === q.correct) {
                clickedBtn.classList.add('correct');
                score++;
                resultStatus = "CORRECTA";
            } else {
                clickedBtn.classList.add('incorrect');
                buttons[q.correct].classList.add('correct');
            }

            answersLog.push({ questionId: currentQuestionIdx + 1, selection: clickedBtn.innerText, result: resultStatus });
            document.getElementById("next-btn").disabled = false;
        }

        function nextQuestion() {
            currentQuestionIdx++;
            if (currentQuestionIdx < quizData.length) {
                loadQuestion();
            } else {
                buildAndShowReport();
            }
        }

        function buildAndShowReport() {
            isQuizEnded = true;
            document.getElementById("screen-quiz").classList.remove("active");
            document.getElementById("screen-finish").classList.add("active");

            const calculatedGrade = score; 

            document.getElementById("form-code").value = currentCode;
            document.getElementById("form-correct").value = score;
            document.getElementById("form-incorrect").value = quizData.length - score;
            document.getElementById("form-grade").value = `${calculatedGrade} / 4 Puntos Totales`;
            
            let logString = "";
            answersLog.forEach(l => {
                logString += `Pregunta ${l.questionId} -> Marcó: ${l.selection} | Estado: ${l.result}\n`;
            });
            document.getElementById("form-details").value = logString;
        }

        function handleFormSubmit(event) {
            const form = event.target;
            if(form.action.includes("https://formspree.io/f/mbdvlkjl")) {
                event.preventDefault();
                alert(`[Modo Prueba Docente Activo]\nCódigo: ${currentCode}\nNota: ${document.getElementById("form-grade").value}\n\n¡Simulación correcta! Recordá poner tu token real de Formspree para recibir los correos en clase.`);
                document.getElementById("screen-finish").classList.remove("active");
                document.getElementById("screen-exit").classList.add("active");
            }
        }
